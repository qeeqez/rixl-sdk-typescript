# @rixl/sdk-typescript-feeds@2.0

A TypeScript SDK client for the localhost API.

## Usage

First, install the SDK from npm.

```bash
npm install @rixl/sdk-typescript-feeds --save
```

Next, try it out.


```ts
import {
  Configuration,
  FeedsApi,
} from '@rixl/sdk-typescript-feeds';
import type { FeedsFeedIdCreatorsCreatorIdGetRequest } from '@rixl/sdk-typescript-feeds';

async function example() {
  console.log("🚀 Testing @rixl/sdk-typescript-feeds SDK...");
  const api = new FeedsApi();

  const body = {
    // string | Feed ID
    feedId: F4edRI23XL,
    // string | Creator ID
    creatorId: C6rtRI23XL,
    // number | Maximum number of items to return in a single request. <br> **Default:** `25` (optional)
    limit: 25,
    // number | Starting point of the result set. <br>To get page 2 with a limit of 25, set `offset` to `25`. <br> **Default:** `0` (optional)
    offset: 0,
  } satisfies FeedsFeedIdCreatorsCreatorIdGetRequest;

  try {
    const data = await api.feedsFeedIdCreatorsCreatorIdGet(body);
    console.log(data);
  } catch (error) {
    console.error(error);
  }
}

// Run the test
example().catch(console.error);
```
